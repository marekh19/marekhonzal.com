Let's see how to implement the component above in **Preact**.

I love `preact`. Let's have a look below. It's just React

```tsx
import { useState } from 'preact/hooks'

export const Counter = () => {
  const [count, setCount] = useState(0)

  const handleIncrement = () => {
    setCount((prev) => ++prev)
  }

  return (
    <button className="btn btn-primary" onClick={handleIncrement}>
      Count: {count}
    </button>
  )
}
```

```ts
import type { QueryResult } from '@apollo/client'

import {
  MiniGameCategory,
  useCatalogMethodologiesQuery,
  useCatalogMiniGamesQuery,
  useCatalogVideoExplainersQuery,
} from '@/__generated__/graphql'

import { catalogFiltersToQueryInput } from '../lib/catalogFiltersToQueryInput'

export type Filters = {
  subjects?: string[]
  themes?: string[]
  difficulties?: string[]
  search?: string
}

const getDataOrPrevious = <T>(queryResult: QueryResult<T>) =>
  queryResult.data ?? queryResult.previousData

const getIsInitialLoading = <T>(queryResult: QueryResult<T>) =>
  queryResult.loading && queryResult.previousData === undefined

export const useCatalog = (filters: Filters) => {
  const variables = catalogFiltersToQueryInput(filters)

  const methodologiesQuery = useCatalogMethodologiesQuery({ variables })
  const videoExplainersQuery = useCatalogVideoExplainersQuery({ variables })
  const miniGamesQuery = useCatalogMiniGamesQuery({
    variables: { ...variables, categories: [MiniGameCategory.Minigame] },
  })
  const eduBooksQuery = useCatalogMiniGamesQuery({
    variables: { ...variables, categories: [MiniGameCategory.Edubook] },
  })

  const methodologies =
    getDataOrPrevious(methodologiesQuery)?.catalog.getMethodologies.filter(
      (methodology) => !methodology.isProject,
    ) ?? []
  const projectMethodologies =
    getDataOrPrevious(methodologiesQuery)?.catalog.getMethodologies.filter(
      (methodology) => methodology.isProject,
    ) ?? []
  const videos = getDataOrPrevious(videoExplainersQuery)?.catalog.getVideoExplainers ?? []
  const minigames =
    getDataOrPrevious(miniGamesQuery)?.catalog.getMiniGamesCollection.results ?? []
  const eduBooks =
    getDataOrPrevious(eduBooksQuery)?.catalog.getMiniGamesCollection.results ?? []

  const isMethodologiesInitialLoading = getIsInitialLoading(methodologiesQuery)
  const isVideosInitialLoading = getIsInitialLoading(videoExplainersQuery)
  const isMinigamesInitialLoading = getIsInitialLoading(miniGamesQuery)
  const isEdubooksInitialLoading = getIsInitialLoading(eduBooksQuery)

  return {
    results: {
      methodologies,
      projectMethodologies,
      videos,
      minigames,
      eduBooks,
    },
    isInitialLoading: {
      methodologies: isMethodologiesInitialLoading,
      projectMethodologies: isMethodologiesInitialLoading,
      videos: isVideosInitialLoading,
      minigames: isMinigamesInitialLoading,
      eduBooks: isEdubooksInitialLoading,
    },
    isLoading: {
      methodologies: methodologiesQuery.loading,
      projectMethodologies: methodologiesQuery.loading,
      videos: videoExplainersQuery.loading,
      minigames: miniGamesQuery.loading,
      eduBooks: eduBooksQuery.loading,
    },
  }
}
```
